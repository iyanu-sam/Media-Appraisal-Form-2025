  <script>

    function calculateScore() {

      const fields = [

        'lifeClass', 'dues', 'designTraining', 'centralPrayer',

        'sotd', 'soundcheck', 'activities', 'onlinePresence', 'evangelism'

      ];

      let totalScore = 0;

 

      fields.forEach(field => {

        const value = parseFloat(document.getElementById(field).value);

        totalScore += value;

      });

 

      const resultDiv = document.getElementById('result');

      resultDiv.innerHTML = `<strong>Total Score:</strong> ${totalScore}%`;

    }
